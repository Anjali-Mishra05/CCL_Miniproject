import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";
import { SNSClient, PublishCommand } from "@aws-sdk/client-sns";
import { DynamoDBClient } from "@aws-sdk/client-dynamodb";
import { DynamoDBDocumentClient, PutCommand } from "@aws-sdk/lib-dynamodb";

const REGION = "us-east-1";

const s3 = new S3Client({ region: REGION });
const dynamo = DynamoDBDocumentClient.from(new DynamoDBClient({ region: REGION }));
const sns = new SNSClient({ region: REGION });

const BUCKET = "petmatch-images-ccl";
const TOPIC_ARN = "arn:aws:sns:us-east-1:668353540629:petmatch-rep-alert";

export const handler = async (event) => {

    // ✅ CORS preflight
    if (event.requestContext?.http?.method === "OPTIONS") {
        return {
            statusCode: 200,
            headers: cors(),
            body: null
        };
    }

    try {
        const body = JSON.parse(event.body || "{}");

        // ✅ Basic validation
        if (!body.image) {
            return {
                statusCode: 400,
                headers: cors(),
                body: JSON.stringify({ message: "Image is required" })
            };
        }

        // ✅ 1. Upload image to S3
        const buffer = Buffer.from(body.image, "base64");
        const fileName = `pets/${Date.now()}.jpg`;

        await s3.send(new PutObjectCommand({
            Bucket: BUCKET,
            Key: fileName,
            Body: buffer,
            ContentType: "image/jpeg"
        }));

        const imageUrl = `https://${BUCKET}.s3.${REGION}.amazonaws.com/${fileName}`;

        // ✅ 2. Save to DynamoDB (MATCHING YOUR FRONTEND)
        const petId = Date.now().toString();

        await dynamo.send(new PutCommand({
            TableName: "PetMatch-Pets",
            Item: {
                petId,
                type: body.type,
                species: body.species,
                name: body.name,
                breed: body.breed,
                color: body.color,
                description: body.description,
                location: body.location,
                date: body.date,
                contactPhone: body.contactPhone,
                email: body.email,
                imageUrl,
                createdAt: new Date().toISOString()
            }
        }));

        // ✅ 3. Send ONE email notification (simple)
        await sns.send(new PublishCommand({
            TopicArn: TOPIC_ARN,
            Subject: "🐾 Pet Report Submitted",
            Message: `
Pet Report:

Name: ${body.name}
Type: ${body.type}
Location: ${body.location}
Contact: ${body.contactPhone}

Image: ${imageUrl}
            `
        }));

        return {
            statusCode: 200,
            headers: cors(),
            body: JSON.stringify({
                message: "Pet saved successfully",
                imageUrl
            })
        };

    } catch (err) {
        console.error("ERROR:", err);

        return {
            statusCode: 500,
            headers: cors(),
            body: JSON.stringify({
                message: "Internal Server Error",
                error: err.message
            })
        };
    }
};

// ✅ CORS helper
function cors() {
    return {
        "Access-Control-Allow-Origin": "*",
        "Access-Control-Allow-Headers": "*",
        "Access-Control-Allow-Methods": "OPTIONS,POST,GET"
    };
}
